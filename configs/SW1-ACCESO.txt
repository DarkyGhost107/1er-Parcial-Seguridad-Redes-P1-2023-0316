! =====================================================================
!  SW1-ACCESO | Switch de acceso (VLANs, Port-Security, hardening L2)
!  VLAN Administrativa: 99  | Gestion remota: SVI VLAN 99
! =====================================================================
enable
configure terminal
!
hostname SW1-ACCESO
!
no ip domain-lookup
ip domain-name lab.local
username admin privilege 15 secret Cisco$2025
enable secret Cisco$2025
service password-encryption
!
banner motd #
  ACCESO RESTRINGIDO - SOLO PERSONAL AUTORIZADO
  Todo acceso no autorizado sera registrado.
#
!
! --- VLANs ---
vlan 10
 name VENTAS
vlan 20
 name OPERACIONES
vlan 99
 name ADMINISTRATIVA
!
! --- Gestion remota: SVI en la VLAN Administrativa ---
interface vlan 99
 ip address 20.23.3.194 255.255.255.224
 no shutdown
!
ip default-gateway 20.23.3.193
!
! --- Trunk hacia el router ---
interface g0/1
 switchport mode trunk
 switchport trunk allowed vlan 10,20,99
 switchport nonegotiate
!
! --- Puertos de acceso VLAN 10 con Port-Security ---
interface range f0/1 - 10
 switchport mode access
 switchport access vlan 10
 switchport port-security
 switchport port-security maximum 2
 switchport port-security violation shutdown
 switchport port-security mac-address sticky
 spanning-tree portfast
 spanning-tree bpduguard enable
!
! --- Puertos de acceso VLAN 20 con Port-Security ---
interface range f0/11 - 20
 switchport mode access
 switchport access vlan 20
 switchport port-security
 switchport port-security maximum 2
 switchport port-security violation shutdown
 switchport port-security mac-address sticky
 spanning-tree portfast
 spanning-tree bpduguard enable
!
! --- Apagar interfaces no utilizadas ---
interface range f0/21 - 24
 shutdown
!
! --- Seguridad adicional L2: proteger el Root Bridge legitimo ---
spanning-tree vlan 1,10,20,99 root primary
!
! --- (Recomendado) DHCP Snooping contra DHCP spoofing ---
ip dhcp snooping
ip dhcp snooping vlan 10,20
interface g0/1
 ip dhcp snooping trust
!
! --- SSH ---
crypto key generate rsa modulus 1024
ip ssh version 2
!
line console 0
 login local
 exec-timeout 5 0
 logging synchronous
!
line vty 0 4
 transport input ssh
 login local
 exec-timeout 5 0
!
login block-for 120 attempts 3 within 60
!
end
write memory
